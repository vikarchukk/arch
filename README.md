```
[ / ]
├─[ /arch ]
│ ├─ install - встановлення arch
│ ├─ app - інструкції по встановленю програм
│ ├─ base - базові налаштування
│ │  └─ wifi, fan, audio, backlight, font, cursor
│ ├─ addition - доповнення налаштуваннь
│ │  └─ autostart-wm, battery-alert, bash-set
│ └─ xorg - графічний інтерфейс X
├─[ /program ]
│ ├─ dwm - віконий менеджер
│ ├─ slstatus - статус бар
│ ├─ st - термінал
│ ├─ herbe - сповіщення
│ ├─ vim - редактор файлів
│ ├─ btop - моніторинг системи
│ ├─ qutebrowser - браузер подібний до vim
│ ├─ bluetui - булютуз
│ ├─ steam - ігри
│ ├─[ звук ]
│ │ ├─ pacmixer - реголювання звуку
│ │ ├─ keysound - звук для клавіш
│ │ ├─ mpv - музика/відео
│ │ └─ cava - візулізація звуку
│ └─ rxfetch - інформація про пк
└─[ /docs ]
  ├─ keyboard-shortcuts - комбінації клавіш
  ├─ command - команди
  ├─ file-system - файлова сестема
  └─ TODO

[ dwm ]
├─[ кирування вікнами ]
│ ├─[ теги ]
│ │ ├─ super_l + number - перейти до вибраного тегу
│ │ │  └─ left click on number tag
│ │ ├─ super_l + shift + number - перекинути активне вікно в вибраний тег
│ │ │  └─ right click on number tag
│ │ ├─ super_l + control + number - обєднує перегляд вибраних тегів
│ │ ├─ super_l + control + shift + number - відобразити активне вікно в вибраних тегах
│ │ ├─ super_l + 0 - переглянути всі вікна
│ │ ├─ super_l + shift + 0 - відобразити вікно в всіх тегах
│ │ ├─ super_l + k - перейти на наступний тег
│ │ └─ super_l + j - перейти на задній тег
│ └─[ вікна ]
│   ├─ super_l + shift + k - перейти на наступне вікно
│   ├─ super_l + shift + j - перейти на заднє вікно
│   ├─ super_l + d - закрити вікно
│   ├─ super_l + m - режим монокля відображиння вікон [M]
│   ├─ super_l + n - режим класичний відображення вікон []=
│   │  ├─ super_l + l - збільшити ширину головної зони
│   │  ├─ super_l + h - зменшити ширину головної зони
│   │  ├─ super_l + shift + l - перейти на наступну розкладку вікон
│   │  ├─ super_l + shift + h - перейти на задню розкладку вікон
│   │  └─ super_l + return - перемістити вікно в головну зону
│   ├─ super_l + b - режим вільний відображиння вікон ><>
│   │  ├─ super_l + click on window - перемістити вікно
│   │  └─ super_l + shift + click on window - змінити розмір вікна
│   └─ super_l + shift + space - вести або вивести вікно з схеми вікон
├─[ реголювання ]
│ ├─ cmd + space - змінювати розкладку клавіатури між українською і англійською
│ ├─[ звук ]
│ │ ├─ cmd + + - збільшити звук
│ │ └─ cmd + - - зменшити звук
│ ├─[ яскравість екрану ]
│ │ ├─ cmd + shift + + - збільшити яскравість
│ │ └─ cmd + shift + - - зменшити яскравість
│ ├─[ яскравість клавіатури ]
│ │ ├─ cmd + control + + - збільшити яскравість
│ │ └─ cmd + control + - - зменшити яскравість
│ ├─[ музика ]
│ │ ├─ option/alt + 1-6 - включити
│ │ └─ option/alt + 0 - виключити
│ ├─[ звук клавіш ]
│ │ ├─ cmd + o - включити
│ │ └─ cmd + shift + o - виключити
│ └─[ блютуз ]
│   ├─ cmd + shift + b - включити і bluetui
│   └─ cmd + control + b - виключити
├─[ програми ]
│ ├─ cmd + shift + return - st
│ ├─ cmd + shift + w - qutebrowser
│ ├─ cmd + shift + m - btop
│ ├─ cmd + shift + p - pacmixer
│ ├─ cmd + shift + g - steam
│ └─ control + space - відкрити меню програм
├─ cmd + g - сховати або показати статус бар
├─ cmd + q - вийти з графічного інтерфейсу X
└─ control + c,v - скопіювати, вставити
```
